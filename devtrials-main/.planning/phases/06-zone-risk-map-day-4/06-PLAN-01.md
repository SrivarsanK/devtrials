This is the **browser-only** inner component. It can safely import from `leaflet`.

```tsx
'use client';

import { useEffect, useRef } from 'react';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';
import { ZONE_GEO_DATA } from '@/data/zones';
import { useDashboardStore } from '@/store/dashboardStore';
import type { ZoneReserve } from '@/types';

// Status → Leaflet circle color
const STATUS_COLORS: Record<string, string> = {
  RED:   '#ef4444',
  AMBER: '#f59e0b',
  GREEN: '#22c55e',
};

function buildPopupHtml(zone: ZoneReserve, isPaused: boolean): string {
  const statusColor = STATUS_COLORS[zone.status] || '#6b7280';
  const statusLabel = zone.status === 'RED' ? '🔴 RED' : zone.status === 'AMBER' ? '🟡 AMBER' : '🟢 GREEN';
  const cbStatus = isPaused ? '⏸ Paused' : '▶ Active';

  return `
    <div style="font-family: system-ui; min-width: 200px; color: #e2e8f0; background: #1e293b; border-radius: 8px; padding: 0;">
      <div style="background: ${statusColor}22; border-bottom: 1px solid ${statusColor}44; padding: 10px 12px; border-radius: 8px 8px 0 0;">
        <strong style="font-size:13px; color: #fff">${zone.name}</strong>
        <span style="float: right; font-size: 10px; color: ${statusColor}; font-weight: 700">${statusLabel}</span>
      </div>
      <div style="padding: 10px 12px; font-size: 11px; line-height: 1.8; color: #94a3b8">
        <div>📍 City: <strong style="color:#e2e8f0">${zone.city}</strong></div>
        <div>📋 Policies: <strong style="color:#e2e8f0">${zone.policies.toLocaleString('en-IN')}</strong></div>
        <div>🏦 Reserve: <strong style="color:#e2e8f0">₹${zone.reserveLakhs}L</strong></div>
        <div>⚖️ Liability: <strong style="color:#e2e8f0">₹${zone.liabilityLakhs}L</strong></div>
        <div>🔮 7d Forecast: <strong style="color:#e2e8f0">₹${zone.predictedClaims7dLakhs}L</strong></div>
        <div>🔌 Circuit Breaker: <strong style="color:${isPaused ? '#ef4444' : '#22c55e'}">${cbStatus}</strong></div>
      </div>
    </div>
  `;
}

export function LeafletMapInner() {
  const mapRef = useRef<L.Map | null>(null);
  const circlesRef = useRef<Record<string, L.Circle>>({});
  const reserveHealth = useDashboardStore((s) => s.reserveHealth);
  const pausedZones = useDashboardStore((s) => s.pausedZones);

  // Initialize map once
  useEffect(() => {
    if (mapRef.current) return;

    const map = L.map('leaflet-map-container', {
      center: [19.5, 78.8],   // center of India
      zoom: 5,
      zoomControl: true,
      attributionControl: false,
    });

    L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
      maxZoom: 14,
    }).addTo(map);

    mapRef.current = map;

    return () => {
      map.remove();
      mapRef.current = null;
    };
  }, []);

  // Render/update circles whenever reserveHealth or pausedZones changes
  useEffect(() => {
    const map = mapRef.current;
    if (!map || !reserveHealth) return;

    // Build flat zone lookup from reserveHealth
    const zoneMap: Record<string, ZoneReserve> = {};
    for (const city of reserveHealth.cities) {
      for (const zone of city.zones) {
        zoneMap[zone.id] = zone;
      }
    }

    ZONE_GEO_DATA.forEach((geo) => {
      const zone = zoneMap[geo.id];
      if (!zone) return;

      const color = STATUS_COLORS[zone.status] || '#6b7280';
      const isPaused = pausedZones.has(zone.name);

      if (circlesRef.current[geo.id]) {
        // Update existing circle style
        circlesRef.current[geo.id].setStyle({ color, fillColor: color });
        circlesRef.current[geo.id].unbindPopup();
        circlesRef.current[geo.id].bindPopup(buildPopupHtml(zone, isPaused), {
          maxWidth: 260,
          className: 'shield-popup',
        });
      } else {
        // Create new circle
        const circle = L.circle([geo.lat, geo.lng], {
          radius: geo.radiusMeters,
          color,
          fillColor: color,
          fillOpacity: 0.35,
          weight: 2,
        })
          .addTo(map)
          .bindPopup(buildPopupHtml(zone, isPaused), {
            maxWidth: 260,
            className: 'shield-popup',
          });

        circlesRef.current[geo.id] = circle;
      }
    });
  }, [reserveHealth, pausedZones]);

  return <div id="leaflet-map-container" style={{ width: '100%', height: '420px', borderRadius: '8px' }} />;
}
```

```tsx
'use client';

import dynamic from 'next/dynamic';
import { MapPin } from 'lucide-react';

// Critical: ssr:false prevents "window is not defined" crash (ZRM-001)
const LeafletMapInner = dynamic(
  () => import('./LeafletMapInner').then((m) => m.LeafletMapInner),
  {
    ssr: false,
    loading: () => (
      <div className="w-full h-[420px] bg-shield-surface rounded-lg flex items-center justify-center">
        <div className="flex flex-col items-center gap-3 text-shield-muted">
          <MapPin className="w-8 h-8 animate-pulse" />
          <span className="text-sm">Loading Zone Risk Map...</span>
        </div>
      </div>
    ),
  }
);

export function ZoneRiskMap() {
  return (
    <div className="bg-shield-card rounded-lg border border-shield-border/50 p-4">
      <div className="flex items-center justify-between mb-3">
        <div className="flex items-center gap-2">
          <MapPin className="w-4 h-4 text-shield-primary" />
          <h3 className="text-white font-semibold text-sm">Zone Risk Map</h3>
        </div>
        <div className="flex items-center gap-3 text-[10px] font-medium">
          <span className="flex items-center gap-1"><span className="w-2 h-2 rounded-full bg-shield-green inline-block" /> Normal</span>
          <span className="flex items-center gap-1"><span className="w-2 h-2 rounded-full bg-shield-amber inline-block" /> Watch</span>
          <span className="flex items-center gap-1"><span className="w-2 h-2 rounded-full bg-shield-red inline-block" /> Critical</span>
        </div>
      </div>
      <LeafletMapInner />
    </div>
  );
}
```
